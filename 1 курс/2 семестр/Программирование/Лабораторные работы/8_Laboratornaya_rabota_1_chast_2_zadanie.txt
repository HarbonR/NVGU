2. Дан файл, содержащий произвольный текст. Выяснить, чего в нем больше: букв или цифр
//====================================================================================================
#include <stdio.h> /// Библиотека ввода-вывода Си
#include <windows.h> /// Библиотека для работы с системными функциями

int main()
{
    system("chcp 1251"); /// Локализуем русский язык
    //====================================================================================================
    FILE *inputFile; /// Объявляем переменную типа файл
    char inputString[100]; /// Переменная входящая строка
    char outputSring[100]; /// Переменная выходящая строка
    int numberLetters = 0; /// Количество букв
    int numberDigits = 0; /// Количество цифр
    printf ("Введите строку: ");
    gets (outputSring); /// Функция gets считывает символы из стандартного потока ввода до символа новой строки n
    //====================================================================================================
    /// A - Z (65 - 90); a - z (97-122); А - Я (192 - 223); а - я (224 - 255); 0 - 9 (48 - 57)
    char englishCapitalLetters[27]; /// Переменная строка хранит большие английские буквы и тернирующий ноль в конце
    /// Заполняем массив большими английскими буквами
    for (int i = 65; i < 91; i++)
        englishCapitalLetters[i - 65] = (char)i;
    englishCapitalLetters[26] = '\0'; /// Присваиваем тернирующий ноль в конец массива
    //==========
    char englishSmallLetters[27]; /// Переменная строка хранит маленькие английские буквы и тернирующий ноль в конце
    /// Заполняем массив маленькими английскими буквами
    for (int i = 97; i < 123; i++)
        englishSmallLetters[i-97] = (char)i;
    englishSmallLetters[26] = '\0'; /// Присваиваем тернирующий ноль в конец массива
    //==========
    char russianletters[67];
    /// Заполняем массив большими и маленькими русскими буквами
    for (int i = 192; i < 256; i++)
        russianletters[i - 192] = (char)i;
    russianletters[66] = '\0'; /// Присваиваем тернирующий ноль в конец массива
    //==========
    char digits[] = "0123456789"; /// Переменная строка хранит числа от 0 до 9, в конце тернирующий ноль
    //====================================================================================================
    /// Открываем файл и записывем в файл строку outputSring
    inputFile = fopen ("inputFile.txt", "wt"); /// Открываем текстовый файл для записи
    if (inputFile == NULL)
    {
        printf ("Ошибка открытия файла");
    }
    else
    {
        fputs (outputSring, inputFile); /// Функция fputs записывает строку, указанную в параметре string в поток filestream
        fclose(inputFile); /// Закрываем файл
    }
    //====================================================================================================
    /// Открываем и считываем из файла строку. Помещаем строку в переменную inputString
    inputFile = fopen ("inputFile.txt", "rt"); /// Открываем текстовый файл для чтения
    if (inputFile == NULL)
    {
        printf ("Ошибка открытия файла");
    }
    else
    {
        fgets (inputString, 100, inputFile); /// Функция fgets считывает символы из потока и сохраняет их в виде строки в параметр string до тех пор пока не наступит конец строки или пока не будет достигнут конец файла.
        fclose(inputFile); /// Закрываем файл
    }
    //====================================================================================================
    /// Пробегаемся по строке. Определяем присутсвие больших английских букв в строке.
    for (int i = 0; inputString[i] != '\0'; i++)
        for (int j = 0; englishCapitalLetters[j] != '\0'; j++)
            if (inputString[i] == englishCapitalLetters[j])
                numberLetters++;
    /// Пробегаемся по строке. Определяем присутсвие маленьких английских букв в строке.
    for (int i = 0; inputString[i] != '\0'; i++)
        for (int j = 0; englishSmallLetters[j] != '\0'; j++)
            if (inputString[i] == englishSmallLetters[j])
                numberLetters++;
    /// Пробегаемся по строке. Определяем присутсвие русских букв в строке.
    for (int i = 0; inputString[i] != '\0'; i++)
        for (int j = 0; russianletters[j] != '\0'; j++)
            if (inputString[i] == russianletters[j])
                numberLetters++;
    /// Пробегаемся по строке. Определяем присутсвие цифр в строке.
    for (int i = 0; inputString[i] != '\0'; i++)
        for (int j = 0; digits[j] != '\0'; j++)
            if (inputString[i] == digits[j])
                numberDigits++;
    //====================================================================================================
    if (numberLetters > numberDigits)
        printf ("Количество букв: %d > Количество цифр: %d", numberLetters, numberDigits);
    else if (numberLetters < numberDigits)
        printf ("Количество букв: %d < Количество цифр: %d", numberLetters, numberDigits);
    else if (numberLetters == numberDigits)
        printf ("Количество букв: %d = Количество цифр: %d", numberLetters, numberDigits);
    //====================================================================================================
    return 0;
}
