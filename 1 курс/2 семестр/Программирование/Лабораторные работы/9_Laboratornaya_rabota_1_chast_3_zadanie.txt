#include <stdio.h> /// Библиотека ввода-вывода Си
#include <stdlib.h> /// Стандартная библиотека
#include <graphics.h> /// Библиотека с графикой

void olympicRings ()
{
    setcolor(BLACK); /// Установить цвет
    //==================================================
    /// Голубой
    setlinestyle(0,0,3); /// Установка ширины и стиля линии (стиль линии, паттерн, толщина линии)
    arc (100,100,-70,275,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (100,100,-60,280,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //==================================================
    /// Черный
    setlinestyle(0,0,3); /// Установка ширины и стиля линии (стиль линии, паттерн, толщина линии)
    arc (210,100,-70,180,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (210,100,-60,180,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //=====
    arc (210,100, 195,275,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (210,100, 205,280,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //==================================================
    /// Красный
    setlinestyle(0,0,3); /// Установка ширины и стиля линии (стиль линии, паттерн, толщина линии)
    arc (320,100,-165,180,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (320,100,-155,180,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //==================================================
    /// Желтый
    setlinestyle(0,0,3); /// Установка ширины и стиля линии (стиль линии, паттерн, толщина линии)
    arc (155,150,15,95,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (155,150,25,95,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //=====
    arc (155,150,110,360,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (155,150,125,360,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //==================================================
    /// Зеленый
    setlinestyle(0,0,3); /// Установка ширины и стиля линии (стиль линии, паттерн, толщина линии)
    arc (265,150,15,95,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (265,150,25,95,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //=====
    arc (265,150,110,360,50); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    arc (265,150,125,360,40); /// Дуга окружности (x, y, начальный угол, конечный угол, радиус)
    //==================================================
    /// Голубой покраска
    setfillstyle(1,LIGHTBLUE); /// Устанавливает цвет и шаблон заполнения (шаблон, цвет)
    floodfill(100, 55, BLACK); /// Заполняет ограниченную область (x, y, цвет границы области)
    //==================================================
    /// Черный покраска
    setfillstyle(1,BLACK); /// Устанавливает цвет и шаблон заполнения (шаблон, цвет)
    floodfill(210, 55, BLACK); /// Заполняет ограниченную область (x, y, цвет границы области)
    floodfill(210, 145, BLACK); /// Заполняет ограниченную область (x, y, цвет границы области)
    //==================================================
    /// Красный покраска
    setfillstyle(1,LIGHTRED); /// Устанавливает цвет и шаблон заполнения (шаблон, цвет)
    floodfill(320, 55, BLACK); /// Заполняет ограниченную область (x, y, цвет границы области)

    setfillstyle(1,LIGHTGREEN); /// Устанавливает цвет и шаблон заполнения (шаблон, цвет)

    floodfill(265, 105, BLACK); /// Заполняет ограниченную область (x, y, цвет границы области)
    //==================================================
}

int main()
{
    initwindow(800,600); /// Открыть окно для графики размером 800 на 600 пикселей
    floodfill(1,1,WHITE); /// Заполняет ограниченную область (x, y, цвет границы области)
    olympicRings();
    getch(); /// Функция ожидает ввода символа без отображения в консоли (Пауза)
    closegraph(); /// Закрыть окно с графикой
    return 0;
}
