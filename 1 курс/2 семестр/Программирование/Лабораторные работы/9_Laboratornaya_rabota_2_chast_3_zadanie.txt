Запрограммировать движение объекта 3 – квадрат.
//====================================================================================================
#include <stdio.h> /// Библиотека ввода-вывода Си
#include <stdlib.h> /// Стандартная библиотека
#include <graphics.h> /// Библиотека с графикой

void motionSquare (int startX, int startY, int stepX, int stepY, int endX, int endY)
{
int i, j;
    for (i = 0, j = 0; abs(i) < endX || abs (j) < endY; i+=stepX, j+=stepY)
    {
        setcolor(BLACK); /// Установить красный цвет
        rectangle(startX + i, startY + j, startX + 40 + i, startY + 40 + j);
        if (kbhit()) break; /// Если будет нажата любая клавиша, то произойдет выход из цикла (kbhit() возвращает ненулевое значение, если была нажата любая клавиша ).
        delay(10); /// Подождать 1 миллисекунду
        setcolor(WHITE); /// Установить белый цвет.
        rectangle(startX + i, startY + j, startX + 40 + i, startY + 40 + j);
    }
}

int main()
{
    initwindow(800,600); /// Открыть окно для графики размером 800 на 600 пикселей
    floodfill(1,1,WHITE); /// Заполняет ограниченную область (x, y, цвет границы области)
    while (!kbhit())   /// цикл будет продолжаться, пока не будет нажата какая-либо клавиша
    {
        motionSquare (700, 450, -1, -1, 100, 0);
        motionSquare (600, 350, -1, +1, 100, 0);
        motionSquare (500, 450, -1, -1, 100, 0);
        motionSquare (400, 350, -1, +1, 100, 0);
    }
    getch(); /// Функция ожидает ввода символа без отображения в консоли (Пауза)
    closegraph(); /// Закрыть окно с графикой
    return 0;
}
