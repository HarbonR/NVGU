6 Лабораторная работа. 1 часть 18 задание
18. Дана строка символов, среди которых есть двоеточия (:). Подсчитать кол-во символов между двумя двоеточиями.
//====================================================================================================
#include <stdio.h> /// Стандартная библиотека Си ввода-вывода
#include <windows.h> /// Библиотека для работы с системными функциями

int main()
{
    system ("chcp 1251 > NULL"); /// Устанавливаем кодовую страницу 1251 (Локализуем русский язык) (Устанавливаем кодировку windows-1251)
    char str[100];
    int indexFirstColon = -1; /// Индекс первого двоеточия
    int indexSecondColon = -1; /// Индекс второго двоеточия
    printf ("Введите строку: ");
    gets (str); /// Функция gets считывает символы из стандартного потока ввода до символа новой строки n
    /// Находим индексы первого и второго двоеточия и присваиваем их соответсвующим переменным
    for (int i = 0; str[i] != '\0'; i++) /// Пока не конец строки выполнять. '\0' - Тернирующий ноль обозначает конец строки
    {
        if (str[i] == ':' && indexFirstColon == -1)
            indexFirstColon = i;
        else if (str[i] == ':' && indexSecondColon == -1)
            indexSecondColon = i;
    }
    printf ("Количество символов между двумя двоеточиями = %d", indexSecondColon - indexFirstColon - 1);
    return 0;
}
