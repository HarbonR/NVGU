3. Дано число n. Вычислите сумму: 1! + 2! + 3! + ... +n!, создав функцию вычисления факториала числа.
//====================================================================================================
#include <stdio.h> /// Стандартная библиотека Си ввода-вывода
#include <locale.h> /// Библиотека для локализации языка

/// Функция нахождения факториала числа
int factorial (int number)
{
    int multiply; /// Переменная для хранения промежуточных результатов вычисления факториала
    if (number == 0)
        return 1; /// Возвращаем 1 т.к факториал 0 = 1
    multiply = number; /// Присваиваем переменной multiply значения переменной number
    number--; /// Дикриментим переменную (уменьшаем значение переменной на 1)
    while (number != 0) /// Пока не равно 0 выполнять
    {
        multiply *= number; /// Умножаем текущее значение переменной multiply на переменную number и присваиваем результат в переменную multiply
        number--; /// Дикриментим переменную (уменьшаем значение переменной на 1)
    }
    return multiply; /// Возвращаем результат
}

int main()
{
    setlocale (LC_ALL, "Rus"); /// Локализуем русский язык
    int n;
    int sum = 0;
    printf ("Введите число n = ");
    scanf ("%d", &n);
    printf ("Сумма: ");
    for (int i = 1; i <= n; i++)
        printf("%d! \t", i);
    printf("\nСумма: ");
    /// Вычисляем сумму факториалов от 1 до n
    for (int i = 1; i <= n; i++)
    {
        printf ("%d\t", factorial(i)); /// Вычисляем факториал числа i и выводим результат в консоль
        sum += factorial(i); /// Складываем текущее значение переменной sum с факториалом числа i и присваиваем результат в переменную sum
    }
    printf ("\nСумма = %d", sum);
    return 0;
}
