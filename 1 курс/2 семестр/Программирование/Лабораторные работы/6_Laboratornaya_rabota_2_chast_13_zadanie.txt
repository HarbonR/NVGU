6 Лабораторная работа. 2 часть 13 задание
13. Дана строка, среди символов которой есть двоеточия. Получить все символы, расположенные между первым и вторым 
двоеточиями. Если второго двоеточия нет, то получить все символы после первого двоеточия.
//====================================================================================================
#include <stdio.h> /// Стандартная библиотека Си ввода-вывода
#include <windows.h> /// Библиотека для работы с системными функциями

int main()
{
    system ("chcp 1251"); /// Устанавливаем кодовую страницу 1251 (Локализуем русский язык) (Устанавливаем кодировку windows-1251)
    char str[100]; /// Переменная строка
    int indexFirstColon = -1; /// Индекс первого двоеточия
    int indexSecondColon = -1; /// Индекс второго двоеточия
    char changeStr[100]; /// Переменная изменённая строка
    /// Обнуляем переменную изменённая строка
    for (int i = 0; i < 100; i++)
    {
        changeStr[i] = '\0';
    }
    int indexForChageStr = 0; /// Индекс для переменной изменённая
    printf ("Введите строку: ");
    gets (str); /// Функция gets считывает символы из стандартного потока ввода до символа новой строки n
    /// Находим индексы первого и второго двоеточия и присваиваем их соответсвующим переменным
    for (int i = 0; str[i] != '\0'; i++) /// Пока не конец строки выполнять. '\0' - Тернирующий ноль обозначает конец строки
    {
        if (str[i] == ':' && indexFirstColon == -1)
            indexFirstColon = i;
        else if (str[i] == ':' && indexSecondColon == -1)
            indexSecondColon = i;
    }
    /// Если второе двоеточие есть
    if (indexSecondColon != -1)
    {
        for (int i = indexFirstColon + 1; i != indexSecondColon; i++)
        {
            changeStr[indexForChageStr] = str[i];
            indexForChageStr++;
        }
    }
    /// Если второго двоеточия нет
    else if (indexSecondColon == -1)
    {
        for (int i = indexFirstColon + 1; str[i] != '\0'; i++)
        {
            changeStr[indexForChageStr] = str[i];
            indexForChageStr++;
        }
    }
    printf ("%s", changeStr);
    return 0;
}
