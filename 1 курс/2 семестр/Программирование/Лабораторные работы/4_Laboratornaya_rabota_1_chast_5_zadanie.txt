#include <iostream>
#include <stdlib.h>
#include <time.h>

using namespace std;

int main()
{
    setlocale(LC_ALL, "Rus"); /// Локализуем русский язык
    srand(time(NULL)); /// Создаём ключ для рандомайзера
    int *arrayRandomNumbers; /// Создаём ссылку на динамический массив
    int number;
    int counterEvenNumber = 0;
    cout << "Введите количество элементов массива: ";
    cin >> number;
    arrayRandomNumbers = new int[number];
    for (int i = 0; i < number; i++)
        arrayRandomNumbers[i] = rand() % 10 + 1;
    for (int i = 0; i < number; i++)
        cout << arrayRandomNumbers[i] << " ";
    cout << endl;
    for (int i = 0; i < number; i += 2)
        if (arrayRandomNumbers[i] % 2 == 0)
            counterEvenNumber++;
    delete [] arrayRandomNumbers;
    cout << "Четных элементов массива на четных местах = " << counterEvenNumber << endl;
    return 0;
}
