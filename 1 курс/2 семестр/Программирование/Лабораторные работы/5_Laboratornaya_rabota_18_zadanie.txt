18. Написать функцию, вычисляющую сумму арифметической прогрессии. В качестве параметров ей передается первый 
элемент последовательности, разность и кол-во (a, d, n).
//====================================================================================================
#include <stdio.h> /// Стандартная библиотека Си ввода-вывода
#include <locale.h> /// Библиотека для локализации языка

/// Функция, вычисляющая сумму арифметической прогрессии
int sumArithmeticProgression (int a, int d, int n)
{
    return ((2*a+d*(n-1))/2)*n; /// Формула суммы арифметической прогрессии
}

int main()
{
    setlocale (LC_ALL, "Rus"); /// Локализуем русский язык
    int a, d, n;
    printf ("Введите:\nПервый элемент последовательности: ");
    scanf ("%d", &a);
    printf ("Разность: ");
    scanf ("%d", &d);
    printf ("Количество: ");
    scanf ("%d", &n);
    printf ("Сумма арифметической прогрессии = %d", sumArithmeticProgression(a, d, n));
    return 0;
}
