#include <stdio.h> /// Стандартная библиотека ввода-вывода
#include <stdlib.h> /// Стандартная библиотека
#include <locale.h> /// Библиотека для локализации языка
#include <time.h> /// Библиотека для работы со временем

/// Удвоеные нечетные числа из input помещаем в output
void doubleUnevenNumbers (int *input, int lengthInput, int **output, int *lengthOutput)
{
    //==================================================
    /// Определяем количество удвоеных нечетных чисел
    int couter = 0; /// Счетчик для подсчета количества удвоеных нечетных чисел
    for (int i = 0; i < lengthInput; i++)
    {
        if (input[i] % 2 == 0 && input[i] % 4 != 0)
            couter++;
    }
    //==================================================
    *output = (int*)malloc(couter * sizeof(int)); /// Выделяем динамическую память для массива типа int
    //==================================================
    couter = 0;
    /// Заполняем выходной массив удвоеными нечетными числами
    for (int i = 0; i < lengthInput; i++)
    {
        if (input[i] % 2 == 0 && input[i] % 4 != 0)
        {
            (*output)[couter] = input[i];
            couter++;
        }
    }
    //==================================================
    *lengthOutput = couter;
}
