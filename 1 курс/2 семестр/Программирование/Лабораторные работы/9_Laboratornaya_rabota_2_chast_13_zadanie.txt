/// Функция рисует цифру 2
void cabinetNumber (int x, int y)
{
    moveto (x, y); /// Изменение значения текущего положения пера
    lineto (x + 50, y); /// Рисует линию от текущей позиции
    lineto (x + 70, y + 20); /// Рисует линию от текущей позиции
    lineto (x + 70, y + 80); /// Рисует линию от текущей позиции
    lineto (x + 50, y + 100); /// Рисует линию от текущей позиции
    lineto (x + 10, y + 100); /// Рисует линию от текущей позиции
    lineto (x + 10, y + 130); /// Рисует линию от текущей позиции
    lineto (x + 40, y + 130); /// Рисует линию от текущей позиции
    lineto (x + 40, y + 120); /// Рисует линию от текущей позиции
    lineto (x +70, y + 120); /// Рисует линию от текущей позиции
    lineto (x + 70, y + 150); /// Рисует линию от текущей позиции
    lineto (x - 20, y + 150); /// Рисует линию от текущей позиции
    lineto (x - 20, y+ 90); /// Рисует линию от текущей позиции
    lineto (x, y + 70); /// Рисует линию от текущей позиции
    lineto (x + 40, y + 70); /// Рисует линию от текущей позиции
    lineto (x + 40, y + 30); /// Рисует линию от текущей позиции
    lineto (x + 10, y + 30); /// Рисует линию от текущей позиции
    lineto (x + 10, y + 50); /// Рисует линию от текущей позиции
    lineto (x - 20, y + 50); /// Рисует линию от текущей позиции
    lineto (x - 20, y + 20); /// Рисует линию от текущей позиции
    lineto (x, y); /// Рисует линию от текущей позиции
}

void motionCabinetNumber (int startX, int startY, int stepX, int stepY, int endX, int endY)
{
    int i, j;
    for (i = 0, j = 0; abs(i) < endX || abs (j) < endY; i+=stepX, j+=stepY)
    {
        setcolor(BLACK); /// Установить красный цвет
        cabinetNumber(startX + i, startY + j);
        if (kbhit()) break; /// Если будет нажата любая клавиша, то произойдет выход из цикла (kbhit() возвращает ненулевое значение, если была нажата любая клавиша ).
        delay(10); /// Подождать 1 миллисекунду
        setcolor(WHITE); /// Установить белый цвет.
        cabinetNumber(startX + i, startY + j);
    }
}

int main()
{
    initwindow(800,600); /// Открыть окно для графики размером 800 на 600 пикселей
    floodfill(1,1,WHITE); /// Заполняет ограниченную область (x, y, цвет границы области)
    while (!kbhit())   /// цикл будет продолжаться, пока не будет нажата какая-либо клавиша
    {
        motionCabinetNumber (100, 100, +1, 0, 200, 0);
        motionCabinetNumber (300, 100, 0, +1, 0, 200);
        motionCabinetNumber (300, 300, +1, 0, 200, 0);
        motionCabinetNumber (500, 300, 0, -1, 0, 200);
    }
    getch(); /// Функция ожидает ввода символа без отображения в консоли (Пауза)
    closegraph(); /// Закрыть окно с графикой
    return 0;
}
