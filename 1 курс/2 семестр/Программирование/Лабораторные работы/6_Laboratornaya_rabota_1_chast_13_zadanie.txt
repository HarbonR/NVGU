#include <stdio.h> /// Стандартная библиотека Си ввода-вывода
#include <windows.h> /// Библиотека для работы с системными функциями

/// Функция заполняет строку тернирующими нулями (Обнуляет строку)
void fillSrtingTZero (char *str, int lengthStr)
{
    for (int i = 0; i < lengthStr; i++)
        str[i] = '\0';
}

/// Возвращает количество слов заканчивающихся на переменную character
int numberWord (char *str, char character)
{
    char word[30]; /// Объявляем переменную слово
    fillSrtingTZero (word, 30); /// Обнуляем переменную
    int j = 0; /// Переменная индекс для переменной word
    int counter = 0; /// Счетчик
    int lengthStr = 0; /// Переменная длина строки
    /// Проверяем есть ли пробел в конце строки. Если нет добавляем пробел в конец строки
    for (int i = 0; str[i] != '\0'; i++)
        lengthStr++; /// Инкриментим длинну строки
    if (str[lengthStr - 1] != ' ') /// Если в конце нет пробела
        str[lengthStr] = ' '; /// Добавляем пробел в конец строки
    for (int i = 0; str[i] != '\0'; i++) /// Пока не конец строки выполнять. '\0' - Тернирующий ноль (означает конец строки)
    {
        if (str[i] == ' ') /// Если символ под индексом i равен пробелу выполнять
        {
            if (word[j - 1] == character) /// Если последний символ слова равен переменной character
                counter++; /// Инкриментим счетчик (Увеличиваем значение переменной на 1)
            j = 0; /// Обнуляем переменную
            fillSrtingTZero (word, 30); /// Обнуляем строку
        }
        else
        {
            word[j] = str[i]; /// Присваиваем переменной word под индексом j значение переменной str под индексом i
            j++; /// Инкриментим переменную
        }
    }
    return counter; /// Возвращаем значение счётчика
}

int main()
{
    system ("chcp 1251"); /// Устанавливаем кодовую страницу 1251 (Локализуем русский язык) (Устанавливаем кодировку windows-1251)
    char str [100];
    char character;
    printf ("Введите строку: ");
    gets (str); /// Функция gets считывает символы из стандартного потока ввода до символа новой строки n
    printf ("Введите символ: ");
    character = getchar(); ///Функция getchar возвращает следующий символ из стандартного потока ввода с отображением символа в консоли
    printf("Количество слов заканчивается буквой %c = %d", character, numberWord (str, character));
    return 0;
}
